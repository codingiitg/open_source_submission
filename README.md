### C++ Code to Output sum of ACSII value of your FullName :-
```c
 #include <iostream>
#include <string>
 using namespace std;

  
int main()
{
    string name;
    cout<<"Enter your full name :- ";
    getline(cin,name);
    int  sum_ascii = 0;

    for(int i =0;i<name.length();i++)
    {
        if(name[i]!=' ')
        {
            sum_ascii +=(int)name[i];  
        }
    }
  cout<<"SUM of ASCII values of your FULL NAME is "<<sum_ascii<<"\n";


    return 0;
}
  
```
