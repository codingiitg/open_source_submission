### Program to output sum of ASCII values of name entered:

```C++
#include<iostream>
#include<string.h>

using namespace std;

int main(void){

    string name;
    cout<<"Please enter your name:"<<endl;

    getline(cin,name);
    int n=name.length();
    int sum=0;

    for (int i = 0; i < n; i++)
    {
        if(name[i]!=' '){
            sum+=name[i];
        }
    }
    cout<<sum;
}

```
