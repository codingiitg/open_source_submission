## Program in C++ to find sum of ASCII values of my name
``` C++
#include <iostream>
#include <string.h>

using namespace std;

int main() {
    char name[]="G Yuvan Shankar";
    int sum=0;
    for(int i=0;i< strlen(name);i++)
    {
        if(name[i]==' ')
            continue;
        else
        sum+= (int)(name[i]);
    }
    cout<<"The sum of ASCII values of my name= "<<sum<<endl;
}
```
